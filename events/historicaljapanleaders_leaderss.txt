add_namespace = historical_japan_leaders

country_event = {
	id = historical_japan_leaders.1
	title = historical_japan_leaders.1.t
	desc = historical_japan_leaders.1.desc
	picture = GFX_report_event_jap_imperial_family
	
	fire_only_once = yes

	trigger = {
			tag = JAP
			JAP = { has_government = fascism } 
			date > 1936.03.09
	        has_country_leader = {
	         character = JAP_keisuke_Okada
	         ruling_only = yes
            }
	}

	mean_time_to_happen = { days = 1 }
	
	option = {
		name = historical_japan_leaders.1.a
		retire_character = JAP_keisuke_Okada
		add_country_leader_role = {
         Character = JAP_koki_hirota
         Promote_leader = yes
         Country_leader = {
          ideology = fascism_ideology
          expire = "1965.1.1.1"
         }
        }
	}
}
