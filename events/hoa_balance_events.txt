# The Fall of China
news_event = {
	id = hoa.start.event.2
	title = hoa.start.event.2.t
	desc = hoa.start.event.2.d
	picture = GFX_news_event_064
	
	hidden = yes
	
	trigger = {
		CHI = { surrender_progress > 0.7 }
		JAP = { has_war_with = CHI }
		NOT = { has_global_flag = fall_of_chinaa }
	}
	
	immediate = {
		set_global_flag = fall_of_chinaa
		JAP = {
			annex_country = {
			target = CHI
			}
			annex_country = {
			target = SHX
			}
			annex_country = {
			target = PRC
			}
			annex_country = {
			target = XSM
			}
			annex_country = {
			target = SIK
			}
			annex_country = {
			target = GXC
			}
			annex_country = {
			target = YUN
			}
			add_named_threat = { threat = 30 name = Fall_of_China }
		}
		
	}
	
	mean_time_to_happen = {
		days = 2
	}
	
	option = {
		name = hoa.start.event.2.a
	}
}
