add_namespace = leaderss

country_event = {
	id = leaderss.1
	title = leaderss.1.t
	desc = leaderss.1.desc
	picture = GFX_report_event_indian_parade
	
	fire_only_once = yes

	trigger = {
			tag = RAJ
			RAJ = { has_government = neutrality } 
			date > 1936.04.18
	        has_country_leader = {
	         character = RAJ_lord_willingdon
	         ruling_only = yes
            }
	}

	mean_time_to_happen = { days = 1 }
	
	option = {
		name = leaderss.1.a
		retire_character = RAJ_lord_willingdon
		add_country_leader_role = {
         Character = RAJ_lord_linlithgow
         Promote_leader = yes
         Country_leader = {
          ideology = despotism
          expire = "1965.1.1.1"
         }
        }
	}
}
